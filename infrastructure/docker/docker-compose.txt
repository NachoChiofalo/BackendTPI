
services:
  postgres:
    image: postgres:15
    container_name: tpi-postgres
    environment:
      POSTGRES_DB: tpi_logistica
      POSTGRES_USER: tpi_user
      POSTGRES_PASSWORD: tpi_pass
      POSTGRES_INITDB_ARGS: "--encoding=UTF-8 --lc-collate=C --lc-ctype=C"
      TZ: UTC
      PGTZ: UTC
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./postgres/init-scripts:/docker-entrypoint-initdb.d
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U tpi_user -d tpi_logistica"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - tpi-network

  keycloak:
    image: quay.io/keycloak/keycloak:23.0

      KEYCLOAK_ADMIN_PASSWORD: admin
      KC_HTTP_PORT: 8080
      KC_HOSTNAME_STRICT: false
      KC_HOSTNAME_STRICT_HTTPS: false
    ports:
      - "8080:8080"
    depends_on:
      postgres:
        condition: service_healthy
    networks:
      - tpi-network

      - SPRING_DATASOURCE_USERNAME=tpi_user

      - KEYCLOAK_AUTH_SERVER_URL=http://keycloak:8080
      - LOGGING_LEVEL_ROOT=INFO

    healthcheck:
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost:8090/actuator/health"]
